Schrijf een 2e functie die de eerste functie aanroept, en vervolgens de teruggegeven boodschap in een div element op je pagina toont. Denk aan de DOM methods!



    <input id="nameInput" onchange="nameChecker()" placeholder="name checker">

    function nameChecker() {
        let nameVar = document.getElementById('nameInput').value;

        let nameCheckerDisplay;

        switch (nameVar) {
            case "Ryno":
            case "Joeri":
            case "Kate":
            case "Manon":
                nameCheckerDisplay = "Deze persoon zit in breakoutroom 1";
                console.log("Deze persoon zit in breakoutroom 1");
                break;
            default: console.log("Deze persoon zit niet breakoutroom 1");
                nameCheckerDisplay = "Deze persoon zit niet in breakoutroom 1"
        }

        document.getElementById("nameCheckerOutput").innerHTML = nameCheckerDisplay;
    }
